# ──────────────────────────────────────────────────────────────────────────────
# requirements.serve.txt
# Dependencies for the serving container ONLY (docker/Dockerfile.serve).
#
# This is intentionally lean — no SageMaker SDK, no matplotlib, no seaborn.
# The training image (Dockerfile) uses requirements.txt which includes those.
#
# Key pin: xgboost==3.2.0
#   MUST match the version used during training on SageMaker.
#   XGBoost 3.2.0 introduced a new categorical encoding format (enc::Recode /
#   CatContainer). Loading a 3.2.0 model with any earlier version causes a
#   hard crash. Do not change this without retraining the model.
# ──────────────────────────────────────────────────────────────────────────────

# ── Serving framework ─────────────────────────────────────────────────────────
fastapi>=0.115.0
uvicorn[standard]>=0.30.0

# ── MLflow (model registry access) ───────────────────────────────────────────
mlflow>=2.9.0
# Pin protobuf: MLflow UI / internal gRPC breaks with protobuf 6.x
protobuf>=3.20,<6

# ── Data handling ─────────────────────────────────────────────────────────────
pandas>=2.0.0
numpy>=1.24.0

# ── ML model flavours ─────────────────────────────────────────────────────────
# All three are included so the image works regardless of which model flavour
# is currently promoted to Production in MLflow. If only XGBoost will ever be
# Production, lightgbm and catboost can be removed to shrink the image.

# PINNED — must match the SageMaker training version exactly
xgboost==3.2.0

scikit-learn>=1.3.0   # MLflow pyfunc layer uses sklearn internally
lightgbm>=4.0.0       # Required if a LightGBM model is promoted to Production
catboost>=1.2.0       # Required if a CatBoost model is promoted to Production

# ── AWS SDK ───────────────────────────────────────────────────────────────────
# boto3 is needed at serving time for:
#   - Providing AWS credentials when the container runs on ECS (IAM role auth)
#   - Any future integration with Secrets Manager for RDS credentials
boto3>=1.28.0

# ── Logging ───────────────────────────────────────────────────────────────────
# structlog: structured JSON logging for CloudWatch-compatible log lines.
# JSONRenderer + TimeStamper gives us machine-parseable output on stdout,
# which ECS/CloudWatch routes automatically.
structlog>=23.1.0

# ── Environment / config ─────────────────────────────────────────────────────
python-dotenv>=1.0.0  # No-op in ECS (env vars set at task level); used locally
